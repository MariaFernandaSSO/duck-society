Juntando tudo: construindo um algoritmo do zero

Putting it all together: building an algorithm from scratch

Nos últimos 6 posts, vimos cada peça separadamente. Agora vamos construir algo completo do início ao fim, usando tudo: variáveis, operadores, condicionais, loops e funções.

O problema: sistema de gerenciamento de tarefas (Todo List) via terminal. Nada de frameworks, nada de banco de dados — só lógica pura.

O que o sistema vai fazer

  • Adicionar tarefas
  • Listar tarefas
  • Marcar tarefa como concluída
  • Remover tarefas
  • Sair do programa

Tudo via menu interativo no terminal.

Passo 1: pseudocódigo (planejar antes de codar)

INICIO
  CRIAR lista vazia de tarefas
  REPETIR para sempre:
    MOSTRAR menu (1-adicionar, 2-listar, 3-concluir, 4-remover, 5-sair)
    LER opcao do usuario
    
    SE opcao == 1:
      LER descricao da tarefa
      ADICIONAR na lista
    SENAO SE opcao == 2:
      PARA CADA tarefa na lista:
        MOSTRAR indice, descricao e status
    SENAO SE opcao == 3:
      LER indice da tarefa
      MARCAR como concluida
    SENAO SE opcao == 4:
      LER indice da tarefa
      REMOVER da lista
    SENAO SE opcao == 5:
      SAIR do loop
    SENAO:
      MOSTRAR "Opcao invalida"
FIM

Passo 2: implementação em Java

import java.util.ArrayList;
import java.util.Scanner;

public class TodoList {

    public static void main(String[] args) {
        ArrayList tarefas = new ArrayList<>();
        ArrayList concluidas = new ArrayList<>();
        Scanner scanner = new Scanner(System.in);
        int opcao;

        do {
            exibirMenu();
            opcao = scanner.nextInt();
            scanner.nextLine(); // limpa o buffer

            switch (opcao) {
                case 1:
                    adicionarTarefa(tarefas, concluidas, scanner);
                    break;
                case 2:
                    listarTarefas(tarefas, concluidas);
                    break;
                case 3:
                    concluirTarefa(tarefas, concluidas, scanner);
                    break;
                case 4:
                    removerTarefa(tarefas, concluidas, scanner);
                    break;
                case 5:
                    System.out.println("Até logo!");
                    break;
                default:
                    System.out.println("Opção inválida!");
            }
        } while (opcao != 5);

        scanner.close();
    }

    public static void exibirMenu() {
        System.out.println("\n===== TODO LIST =====");
        System.out.println("1 - Adicionar tarefa");
        System.out.println("2 - Listar tarefas");
        System.out.println("3 - Concluir tarefa");
        System.out.println("4 - Remover tarefa");
        System.out.println("5 - Sair");
        System.out.print("Escolha: ");
    }

    public static void adicionarTarefa(
            ArrayList tarefas,
            ArrayList concluidas,
            Scanner scanner) {

        System.out.print("Descrição da tarefa: ");
        String descricao = scanner.nextLine();
        tarefas.add(descricao);
        concluidas.add(false);
        System.out.println("Tarefa adicionada!");
    }

    public static void listarTarefas(
            ArrayList tarefas,
            ArrayList concluidas) {

        if (tarefas.isEmpty()) {
            System.out.println("Nenhuma tarefa cadastrada.");
            return;
        }

        System.out.println("\n--- Suas tarefas ---");
        for (int i = 0; i < tarefas.size(); i++) {
            String status = concluidas.get(i) ? "[OK]" : "[ ]";
            System.out.println((i + 1) + ". " + status + " " + tarefas.get(i));
        }
    }

    public static void concluirTarefa(
            ArrayList tarefas,
            ArrayList concluidas,
            Scanner scanner) {

        listarTarefas(tarefas, concluidas);

        if (tarefas.isEmpty()) return;

        System.out.print("Número da tarefa para concluir: ");
        int indice = scanner.nextInt() - 1;

        if (indice >= 0 && indice < tarefas.size()) {
            concluidas.set(indice, true);
            System.out.println("Tarefa concluída!");
        } else {
            System.out.println("Índice inválido!");
        }
    }

    public static void removerTarefa(
            ArrayList tarefas,
            ArrayList concluidas,
            Scanner scanner) {

        listarTarefas(tarefas, concluidas);

        if (tarefas.isEmpty()) return;

        System.out.print("Número da tarefa para remover: ");
        int indice = scanner.nextInt() - 1;

        if (indice >= 0 && indice < tarefas.size()) {
            tarefas.remove(indice);
            concluidas.remove(indice);
            System.out.println("Tarefa removida!");
        } else {
            System.out.println("Índice inválido!");
        }
    }
}

Passo 3: implementação em Go

package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    tarefas := []string{}
    concluidas := []bool{}
    reader := bufio.NewReader(os.Stdin)

    for {
        opcao := exibirMenu(reader)

        switch opcao {
        case 1:
            adicionarTarefa(&tarefas, &concluidas, reader)
        case 2:
            listarTarefas(tarefas, concluidas)
        case 3:
            concluirTarefa(&tarefas, &concluidas, reader)
        case 4:
            removerTarefa(&tarefas, &concluidas, reader)
        case 5:
            fmt.Println("Até logo!")
            return
        default:
            fmt.Println("Opção inválida!")
        }
    }
}

func exibirMenu(reader *bufio.Reader) int {
    fmt.Println("\n===== TODO LIST =====")
    fmt.Println("1 - Adicionar tarefa")
    fmt.Println("2 - Listar tarefas")
    fmt.Println("3 - Concluir tarefa")
    fmt.Println("4 - Remover tarefa")
    fmt.Println("5 - Sair")
    fmt.Print("Escolha: ")

    entrada, _ := reader.ReadString('\n')
    entrada = strings.TrimSpace(entrada)
    opcao, _ := strconv.Atoi(entrada)
    return opcao
}

func adicionarTarefa(tarefas *[]string, concluidas *[]bool, reader *bufio.Reader) {
    fmt.Print("Descrição da tarefa: ")
    descricao, _ := reader.ReadString('\n')
    descricao = strings.TrimSpace(descricao)

    *tarefas = append(*tarefas, descricao)
    *concluidas = append(*concluidas, false)
    fmt.Println("Tarefa adicionada!")
}

func listarTarefas(tarefas []string, concluidas []bool) {
    if len(tarefas) == 0 {
        fmt.Println("Nenhuma tarefa cadastrada.")
        return
    }

    fmt.Println("\n--- Suas tarefas ---")
    for i := 0; i < len(tarefas); i++ {
        status := "[ ]"
        if concluidas[i] {
            status = "[OK]"
        }
        fmt.Printf("%d. %s %s\n", i+1, status, tarefas[i])
    }
}

func concluirTarefa(tarefas *[]string, concluidas *[]bool, reader *bufio.Reader) {
    listarTarefas(*tarefas, *concluidas)

    if len(*tarefas) == 0 {
        return
    }

    fmt.Print("Número da tarefa para concluir: ")
    entrada, _ := reader.ReadString('\n')
    entrada = strings.TrimSpace(entrada)
    indice, _ := strconv.Atoi(entrada)
    indice--

    if indice >= 0 && indice < len(*tarefas) {
        (*concluidas)[indice] = true
        fmt.Println("Tarefa concluída!")
    } else {
        fmt.Println("Índice inválido!")
    }
}

func removerTarefa(tarefas *[]string, concluidas *[]bool, reader *bufio.Reader) {
    listarTarefas(*tarefas, *concluidas)

    if len(*tarefas) == 0 {
        return
    }

    fmt.Print("Número da tarefa para remover: ")
    entrada, _ := reader.ReadString('\n')
    entrada = strings.TrimSpace(entrada)
    indice, _ := strconv.Atoi(entrada)
    indice--

    if indice >= 0 && indice < len(*tarefas) {
        *tarefas = append((*tarefas)[:indice], (*tarefas)[indice+1:]...)
        *concluidas = append((*concluidas)[:indice], (*concluidas)[indice+1:]...)
        fmt.Println("Tarefa removida!")
    } else {
        fmt.Println("Índice inválido!")
    }
}

O que cada parte usa dos posts anteriores

  • Variáveis e tipos: String, int, boolean, ArrayList
  • Operadores: ==, !=, <, >=, &&
  • Condicionais: if/else (validando índices), switch (menu)
  • Loops: do-while (loop principal), for (listar tarefas)
  • Funções: cada operação é uma função separada

Pra você praticar

O sistema está funcional, mas dá pra melhorar. Tente implementar:

  • Editar descrição de uma tarefa
  • Prioridade (alta, média, baixa) com filtro por prioridade
  • Data de vencimento e ordenar por data
  • Salvar as tarefas em arquivo e carregar na inicialização
